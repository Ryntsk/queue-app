const tg = window.Telegram.WebApp; const API_URL = "https://твой-проект.onrender.com"; // Ссылка с Render async function updateGrid() { const response = await fetch(`${API_URL}/get_queue`); const data = await response.json(); const grid = document.getElementById('queueGrid'); grid.innerHTML = ''; // Очищаем for (let i = 1; i <= 30; i++) { const div = document.createElement('div'); div.className = 'slot'; if (data[i]) { div.classList.add('occupied'); div.innerHTML = `${i}${data[i]}`; } else { div.innerHTML = `${i}Свободно`; div.onclick = () => { tg.sendData(JSON.stringify({slot: i})); }; } grid.appendChild(div); } } // Обновлять каждые 3 секунды для "живого" эффекта setInterval(updateGrid, 3000); updateGrid();
